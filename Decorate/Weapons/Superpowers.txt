
ACTOR PPGSuperpower1 : CustomInventory
{
	Inventory.PickupMessage "You got a Superpower!"
	Inventory.PickupSound "weapon/pickup"
	+FLOATBOB
	RenderStyle Add
	States
	{
	Spawn:
		SPWR A -1
		Stop
	Pickup:
		TNT1 A 0 A_JumpIfInventory("Backpack",1,2)
		TNT1 A 0 A_JumpIfInventory("PPGEnergy",300,"PickupFail")
		TNT1 A 0 A_JumpIfInventory("PPGEnergy",600,"PickupFail")
	PickupTrue:
		TNT1 A 0 A_JumpIf(CallACS("PlayerClassChecker") == 0,"PPGIceBreath")
		Stop
	PPGIceBreath:
		TNT1 A 0 A_SpawnItem("PPGIceBreath")
		Goto End
	End:
		TNT1 A 0 A_KillMaster
		Stop
	PickupFail:
		TNT1 A 0
		fail
	}
}
